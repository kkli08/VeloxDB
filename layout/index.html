Bloom Filter for each Leaf Page

    struct LeafNodeData {
        std::vector<KeyValueWrapper> keyValues;
        uint64_t nextLeafOffset; // Offset to next leaf node

        // Bloom filter for the leaf node
        BloomFilter bloomFilter;
        bool hasBloomFilter = false;
    }
